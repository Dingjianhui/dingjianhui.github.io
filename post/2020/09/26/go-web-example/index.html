package main

import (
	"fmt"
	"github.com/gorilla/mux"
	"net/http"
)

func main()  {
	r := mux.NewRouter()
	// 路由分组
	bookrouter := r.PathPrefix("/v1").Subrouter()
	bookrouter.HandleFunc("/books", CreateBook()).Methods("POST")
	bookrouter.HandleFunc("/books", ReadBook()).Methods("GET")
	bookrouter.HandleFunc("/books", UpdateBook()).Methods("PUT")
	bookrouter.HandleFunc("/books", DeleteBook()).Methods("DELETE")
	http.ListenAndServe(":8080", bookrouter)
}

func CreateBook() http.HandlerFunc {
	return func(writer http.ResponseWriter, request *http.Request) {
		fmt.Fprintf(writer, "Create Book")
	}
}

func ReadBook() http.HandlerFunc {
	return func(writer http.ResponseWriter, request *http.Request) {
		fmt.Fprintf(writer, "Read Book")
	}
}

func UpdateBook() http.HandlerFunc {
	return func(writer http.ResponseWriter, request *http.Request) {
		fmt.Fprintf(writer, "Update Book")
	}
}

func DeleteBook() http.HandlerFunc {
	return func(writer http.ResponseWriter, request *http.Request) {
		fmt.Fprintf(writer, "Delete Book")
	}
}

package main

import (
	"database/sql"
	"fmt"
	_ "github.com/go-sql-driver/mysql"
	"log"
	"time"
)

func main() {
	// 配置数据库连接
	db, err := sql.Open("mysql", "root:123456@(192.168.137.128:3306)/test?parseTime=true")
	if err != nil {
		log.Fatal("数据库连接失败",err)
	}
	if err := db.Ping(); err != nil { // 验证数据库连接
		log.Fatal(err)
	}

	{ // Create a new table
		query := `
            CREATE TABLE users (
                id INT AUTO_INCREMENT,
                username TEXT NOT NULL,
                password TEXT NOT NULL,
                created_at DATETIME,
                PRIMARY KEY (id)
            );`

		if _, err := db.Exec(query); err != nil {
			log.Fatal("数据库表创建失败",err)
		}
	}

	{ // Insert a new user
		username := "johndoe"
		password := "secret"
		createdAt := time.Now()

		result, err := db.Exec(`INSERT INTO users (username, password, created_at) VALUES (?, ?, ?)`, username, password, createdAt)
		if err != nil {
			log.Fatal("数据插入失败",err)
		}

		id, err := result.LastInsertId()
		fmt.Println("插入成功，用户ID：",id)
	}

	{ // Query a single user
		var (
			id        int
			username  string
			password  string
			createdAt time.Time
		)

		query := "SELECT id, username, password, created_at FROM users WHERE id = ?"
		if err := db.QueryRow(query, 1).Scan(&id, &username, &password, &createdAt); err != nil {
			log.Fatal("查询失败",err)
		}

		fmt.Println("用户详情：",id, username, password, createdAt)
	}

	{ // Query all users
		type user struct {
			id        int
			username  string
			password  string
			createdAt time.Time
		}

		rows, err := db.Query(`SELECT id, username, password, created_at FROM users`)
		if err != nil {
			log.Fatal("查询失败",err)
		}
		defer rows.Close()

		var users []user
		for rows.Next() {
			var u user

			err := rows.Scan(&u.id, &u.username, &u.password, &u.createdAt)
			if err != nil {
				log.Fatal(err)
			}
			users = append(users, u)
		}
		if err := rows.Err(); err != nil {
			log.Fatal(err)
		}

		fmt.Printf("用户列表：%#v   \n", users)
	}

	{
		_, err := db.Exec(`DELETE FROM users WHERE id = ?`, 1)
		if err != nil {
			log.Fatal("删除失败",err)
		}
		log.Println("ID为1的用户删除成功")
	}
}

package main

import (
	"html/template"
	"log"
	"net/http"
)

type ContactDetails struct {
	Email   string
	Subject string
	Message string
}

func main()  {
	tmpl := template.Must(template.ParseGlob("views/form.html"))
	http.HandleFunc("/message", func(writer http.ResponseWriter, request *http.Request) {
		if request.Method != http.MethodPost {
			tmpl.Execute(writer,nil)
			return
		}

		details := ContactDetails{
			Email:   request.FormValue("email"),
			Subject: request.FormValue("subject"),
			Message: request.FormValue("message"),
		}
		log.Println(details)

		tmpl.Execute(writer, struct {
			Success bool
			Email string
		}{true,details.Email})
	})
	http.ListenAndServe(":8080",nil)
}

package main

import (
    "fmt"
    "log"
    "net/http"
)

func logging(f http.HandlerFunc) http.HandlerFunc {
    return func(w http.ResponseWriter, r *http.Request) {
        log.Println(r.URL.Path)
        f(w, r)
    }
}

func foo(w http.ResponseWriter, r *http.Request) {
    fmt.Fprintln(w, "foo")
}

func bar(w http.ResponseWriter, r *http.Request) {
    fmt.Fprintln(w, "bar")
}

func main() {
    http.HandleFunc("/foo", logging(foo))
    http.HandleFunc("/bar", logging(bar))

    http.ListenAndServe(":8080", nil)
}

package main

import (
	"fmt"
	"github.com/gorilla/mux"
	"log"
	"net/http"
	"time"
)

// 高级中间件的使用

func main()  {
	r := mux.NewRouter()
	r.Path("/test").
		Queries("token","{token}").
		HandlerFunc(Apply(HelloService(),AuthMiddleware(),LoggerMiddleware()))
	http.ListenAndServe(":8080",r)
}

type Middleware func(http.HandlerFunc) http.HandlerFunc

func Apply(hf http.HandlerFunc, middlewarees ...Middleware) http.HandlerFunc {
	for _,middleware := range middlewarees {
		hf = middleware(hf)
	}
	return hf
}

// 业务层
func HelloService() http.HandlerFunc {
	return func(writer http.ResponseWriter, request *http.Request) {
		fmt.Fprintln(writer, "hello world")
	}
}

func AuthMiddleware() Middleware {
	// Create Auth Middleware
	return func(hf http.HandlerFunc) http.HandlerFunc {
		return func(writer http.ResponseWriter, request *http.Request) {
			query := mux.Vars(request)
			// Middleware to do somthing
			if query["token"] != "token" {
				http.Error(writer, http.StatusText(http.StatusBadRequest), http.StatusBadRequest)
				return
			}
			// Call the next Middleware/Handler in Apply
			hf(writer,request)
		}
	}
}

func LoggerMiddleware() Middleware {
	// Create Logger Middleware
	return func(hf http.HandlerFunc) http.HandlerFunc {
		return func(writer http.ResponseWriter, request *http.Request) {
			// Middleware to do somthing
			start := time.Now()
			defer func() { log.Println(request.URL.Path, time.Since(start)) }()
			// Call the next Middleware/Handler in Apply
			hf(writer,request)
		}
	}
}

	"log"

func main()  {
	http.HandleFunc("/login",login())  // 登录设置session

	http.ListenAndServe(":8080",nil)
}

var (
	key = []byte("abc")
	store = sessions.NewCookieStore(key)
)

func login() http.HandlerFunc {
	return func(writer http.ResponseWriter, request *http.Request) {
		session,err := store.Get(request,"cookie-name")
		if err != nil {
			log.Println("login",err.Error())
		}

		// todo 登录验证

		// 登录成功
		session.Values["auth"] = true
		session.Save(request,writer)

	}
}

func logout() http.HandlerFunc {
	return func(writer http.ResponseWriter, request *http.Request) {
		session,err := store.Get(request,"cookie-name")
		if err != nil {
			log.Println("logout",err.Error())
		}

		// 登录退出
		session.Values["auth"] = false
		session.Save(request,writer)
	}
}

// 首页需要登录后才能显示
func home() http.HandlerFunc {
	return func(writer http.ResponseWriter, request *http.Request) {
		session,err := store.Get(request,"cookie-name")
		if err != nil {
			log.Println("home",err.Error())
		}

		// 验证是否登录,未登录则禁止访问
		if auth,ok := session.Values["auth"].(bool); !ok || !auth {
			http.Error(writer, "Forbidden", http.StatusForbidden)
			return
		}

		fmt.Fprintln(writer, "Home Page")
	}
}

        var user User
        json.NewDecoder(r.Body).Decode(&user)

        fmt.Fprintf(w, "%s %s is %d years old!", user.Firstname, user.Lastname, user.Age)
    })

    http.HandleFunc("/encode", func(w http.ResponseWriter, r *http.Request) {
        peter := User{
            Firstname: "John",
            Lastname:  "Doe",
            Age:       25,
        }

        json.NewEncoder(w).Encode(peter)
    })

    http.ListenAndServe(":8080", nil)
}

package main

import (
	"fmt"
	"github.com/gorilla/websocket"
	"net/http"
)

var upgrader = websocket.Upgrader{
	ReadBufferSize:    1024,
	WriteBufferSize:   1024,
}

func main()  {
	http.HandleFunc("/echo", func(writer http.ResponseWriter, request *http.Request) {
		conn,_ := upgrader.Upgrade(writer,request,nil)
		for {
			// Read message from browser
			msgType, msg, err := conn.ReadMessage()
			if err != nil {
				return
			}
			// Print the message to the console
			fmt.Printf("%s sent: %s\n", conn.RemoteAddr(), string(msg))
			// Write message back to browser
			if err = conn.WriteMessage(msgType, msg); err != nil {
				return
			}
		}
	})

	http.HandleFunc("/", func(writer http.ResponseWriter, request *http.Request) {
		http.ServeFile(writer,request,"views/websockets.html")
	})

	http.ListenAndServe(":8080",nil)
}

package main

import (
    "fmt"

    "golang.org/x/crypto/bcrypt"
)

func HashPassword(password string) (string, error) {
    bytes, err := bcrypt.GenerateFromPassword([]byte(password), 14)
    return string(bytes), err
}

func CheckPasswordHash(password, hash string) bool {
    err := bcrypt.CompareHashAndPassword([]byte(hash), []byte(password))
    return err == nil
}

func main() {
    password := "secret"
    hash, _ := HashPassword(password) // ignore error for the sake of simplicity

    fmt.Println("Password:", password)
    fmt.Println("Hash:    ", hash)

    match := CheckPasswordHash(password, hash)
    fmt.Println("Match:   ", match)
}
